# 1. Create the synthrealm directory if it doesn't exist
sudo mkdir -p /usr/local/share/ca-certificates/synthrealm

# 2. Copy the new CA file to the trust store
# (Adjust 'kanidm_ca.pem' to the actual filename you transferred)
sudo cp /path/to/kanidm_ca.pem /usr/local/share/ca-certificates/synthrealm/kanidm-internal-ca.crt

# 3. Update the system trust store
sudo update-ca-certificates
